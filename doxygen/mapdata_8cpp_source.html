507 void MapData::DrawScaleBar(double width,
508  double height,
509  QString units,
510  QColor color,
511  QString position,
512  QFont font)
513 {
514  //instantiate shape
515 
516  //QCustomPlot object will take ownership of these and make sure they're deleted.
517  QCPItemRect *scale_bar = new QCPItemRect(map_qcp_);
518  QCPItemText *scale_bar_text = new QCPItemText(map_qcp_);
519 
520 
521  map_qcp_->addItem(scale_bar);
522  map_qcp_->addItem(scale_bar_text);
523 
524  QString text = QString::number(width) + " " + units;
525  scale_bar_text->setText(text);
526  scale_bar_text->setPositionAlignment(Qt::AlignCenter);
527  scale_bar_text->setColor(color);
528  scale_bar_text->setFont(font);
529 
530  QPen pen(color);
531  QBrush brush(color, Qt::SolidPattern);
532  scale_bar->setPen(pen);
533  scale_bar->setBrush(brush);
534  QCPRange key_range = map_->keyAxis()->range();
535  QCPRange value_range = map_->valueAxis()->range();
536 
537  //find corners of image
538  double key_max = key_range.upper;
539  double key_min = key_range.lower;
540  double value_max = value_range.upper;
541  double value_min = value_range.lower;
542 
543  double x_min, y_min, x_max, y_max;
544 
545  if (position == "Top Left"){
546  x_min = key_min + 2.0;
547  x_max = x_min + width;
548  y_max = value_max - 4.0;
549  y_min = y_max - height;
550  }
551 
552  else if (position == "Top Right"){
553  x_max = key_max - 2.0;
554  x_min = x_max - width;
555  y_max = value_max - 4.0;
556  y_min = y_max - height;
557  }
558 
559  else if (position == "Bottom Left"){
560  x_min = key_min + 2.0;
561  x_max = x_min + width;
562  y_min = value_min + 2.0;
563  y_max = y_min + height;
564  }
565 
566  else{
567  x_max = key_max - 2.0;
568  x_min = x_max - width;
569  y_min = value_min + 2.0;
570  y_max = y_min + height;
571  }
572 
573  scale_bar->bottomRight->setCoords(x_max, y_min);
574  scale_bar->topLeft->setCoords(x_min, y_max);
575  double x_mid = ((x_max - x_min) / 2.0) + x_min;
576  double y_text = y_max + 1;
577  if (font.pointSize() >= 10){
578  y_text += (font.pointSize() - 10) / 4;
579  }
580 
581  scale_bar_text->position->setCoords(x_mid, y_text);
582 
583  map_qcp_->show();
584 }
