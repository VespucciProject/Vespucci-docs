297 void MainWindow::on_actionNormalize_Standardize_triggered()
298 {
299  TreeItem *item = dataset_tree_model_->getItem(dataset_tree_view_->currentIndex());
300  if (item->type() == TreeItem::ItemType::Base){
301  QMessageBox::information(this,
302  "No datasets loaded",
303  "No dataset exists on which to perform this operation");
304  return;
305  }
306  QString dataset_key = item->DatasetKey();
307  QSharedPointer<VespucciDataset> dataset = workspace->GetDataset(dataset_key);
308  QStringList methods;
309  methods << "Min/Max" << "Vector Norm" << "Unit Area" << "Z-score"
310  << "Standard Normal Variate" << "Peak Intensity" << "Scale Spectra"
311  << "Absolute Value" << "Mean Center";
312  bool ok;
313  QString method = QInputDialog::getItem(this,
314  "Normalization/Standardization",
315  "Method:", methods, 0, false, &ok);
316  double scaling_factor, offset;
317  try{
318  if (ok && method == "Min/Max"){dataset->MinMaxNormalize();}
319  else if (ok && method == "Vector Norm"){dataset->VectorNormalize();}
320  else if (ok && method == "Mean Center"){dataset->MeanCenter();}
321  else if (ok && method == "Unit Area"){dataset->UnitAreaNormalize();}
322  else if (ok && method == "Z-score"){
323  offset = QInputDialog::getDouble(this, "Enter Offset", "Offset",
324  0, -2147483647, 2147483647, 4, &ok);
325  dataset->SNVNormalize(offset, true);
326  }
327 
328  else if (ok && method == "Peak Intensity"){
329  double min = dataset->wavelength_ptr()->min();
330  double max = dataset->wavelength_ptr()->max();
331  RangeDialog *range_dialog = new RangeDialog(this, min, max);
332  range_dialog->setWindowTitle("Peak Intensity Normalization");
333  QObject::connect(range_dialog, SIGNAL(DialogAccepted(double,double)), this, SLOT(RangeDialogAccepted(double,double)));
334  range_dialog->show();
335  }
336  else if (ok && method == "Scale Spectra"){
337  scaling_factor = QInputDialog::getDouble(this, "Enter Scaling Factor",
338  "Factor", 1, -100, 100, 2, &ok);
339  dataset->Scale(scaling_factor);
340  }
341  else if (ok && method == "Standard Normal Variate"){
342  offset = QInputDialog::getDouble(this, "Enter Offset", "Offset",
343  0, -2147483647, 2147483647, 4, &ok);
344  dataset->SNVNormalize(offset, false);
345  }
346  else if (ok && method == "Absolute Value"){dataset->AbsoluteValue();}
347  else
348  return;
349  }
350 
351  catch(exception e){
352  DisplayExceptionWarning(e);
353  }
354 }

359 void MainWindow::on_actionSubtract_Background_triggered()
360 {
361  TreeItem *item = dataset_tree_model_->getItem(dataset_tree_view_->currentIndex());
362  if (item->type() == TreeItem::ItemType::Base){
363  QMessageBox::information(this,
364  "No datasets loaded",
365  "No dataset exists on which to perform this operation");
366  return;
367  }
368 
369  QString dataset_key = item->DatasetKey();
370 
371  QSharedPointer<VespucciDataset> dataset = workspace->GetDataset(dataset_key);
372  QString filename =
373  QFileDialog::getOpenFileName(this,
374  tr("Select Background File"),
375  workspace->directory(),
376  "Vespucci Spectrum Files (*.arma *.txt *.csv)");
377  mat input;
378  QFileInfo file_info(filename);
379  workspace->set_directory(file_info.dir().path());
380  bool success = input.load(filename.toStdString());
381  //We take spectra inputs as row-major
382  if (input.n_rows < input.n_cols){
383  try{
384  input = input.t();
385  }catch(exception e){
386  DisplayExceptionWarning(e);
387  }
388  }
389  //The first column is probably an abscissa.
390  if (input.n_cols > 2){
391  success = false;
392  }
393  if (input.n_cols == 2){
394  try{
395  input.shed_col(0);
396  }catch(exception e){
397  DisplayExceptionWarning(e);
398  }
399  }
400  if (!success){
401  QMessageBox::warning(this, "File Open Error", "File cannot be opened or contains improper data.");
402  return;
403  }
404  else{
405  try{
406  dataset->SubtractBackground(input, filename);
407  }
408  catch(exception e){
409  DisplayExceptionWarning(e);
410  }
411  }
412 }
