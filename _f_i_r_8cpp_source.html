36 arma::mat Vespucci::Math::Smoothing::sgolay(arma::uword poly_order,
37  arma::uword window_size,
38  arma::uword deriv_order,
39  arma::uword scaling_factor)
40 {
41  if (window_size % 2 == 0){
42  std::cerr << "sgolay: invalid window size. Window size must be odd." << std::endl;
43  std::cerr << "using window size poly_order + 2 or nearest odd value" << std::endl;
44  window_size = poly_order + 3 - (poly_order % 2);
45  }
46 
47  if (deriv_order > poly_order){
48  std::cerr << "sgolay: derivative order greater than polynomial order." << std::endl;
49  std::cerr << "using derivative order poly order - 1" << std::endl;
50  deriv_order = poly_order - 1;
51  }
52  arma::mat F = arma::zeros(window_size, window_size);
53  arma::uword k = (window_size - 1) / 2;
54  arma::mat C(window_size, poly_order + 1);
55  arma::sword signed_window_size = window_size;
56  arma::vec column;
57  arma::sword row; //starting at 1 not 0
58  arma::mat A;
59  for (arma::uword i = 0; i <= k; ++i){
60  row = i + 1;
61  column = arma::linspace<arma::vec>((1-row), (signed_window_size - row), window_size);
62  for (arma::uword j = 0; j <= poly_order; ++j){
63  C.col(j) = arma::pow(column, j);
64  }
65  A = pinv(C);
66  F.row(i) = A.row(deriv_order);
67  }
68  arma::sword sign = (deriv_order % 2 == 0 ? 1: -1);
69  arma::uword start_index = 0;
70  for (arma::uword i = window_size - 1; i >k; --i){
71  F.row(i) = sign*F.row(start_index);
72  start_index++;
73  }
74  double product = (deriv_order == 0 ? 1 : prod(arma::linspace<arma::vec>(1, deriv_order, deriv_order)));
75  double power = pow(scaling_factor, deriv_order);
76  F /= (power/product);
77  return F;
78 }

126 arma::vec Vespucci::Math::Smoothing::ApplyFilter(const arma::vec &x, arma::mat coefficients, arma::uword window_size)
127 {
128  arma::uword k = (window_size - 1) / 2;
129  arma::vec filter = trans(coefficients.row(k));
130  //coefficients output by sgolay are in reverse order expected.
131  filter = fliplr(filter);
132  arma::vec filtered_data = arma::conv(x, filter, "same");
133  //Now we have to fix the ends of the filtered data
134  filtered_data.rows(0, k) = coefficients.rows(k, window_size-1)*x.rows(0, window_size-1);
135  filtered_data.rows(x.n_rows - k, x.n_rows - 1) = coefficients.rows(0, k - 1)*x.rows(x.n_rows - window_size, x.n_rows - 1);
136  return filtered_data;
137 }
