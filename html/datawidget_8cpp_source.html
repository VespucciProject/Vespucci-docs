56 void DataWidget::on_forwardPushButton_clicked()
57 {
58  //user sees indexing start at one, table model indexing starts at 0
59  uint next_start_column = current_start_column_ + 15;
60  uint next_end_column = next_start_column + 15;
61  next_end_column = (next_end_column < matrix_columns_ ? next_end_column : matrix_columns_);
62  back_push_button_->setEnabled(true);
63  forward_push_button_->setEnabled(next_end_column - 1 < matrix_columns_);
64  current_start_column_ = next_start_column;
65  table_model_ = new VespucciTableModel(table_view_, table_model_->GetMatrix(), next_start_column-1);
66  table_view_->setModel(table_model_);
67  QString label = "Showing columns " + QString::number(current_start_column_)
68  + "–" + QString::number(next_end_column);
69  column_label_->setText(label);
70 }

72 void DataWidget::on_backPushButton_clicked()
73 {
74  uint next_start_column = current_start_column_ - 15;
75  //since it's unsigned, next_start_column evaluates false if current_start_column_
76  //is less than or equal to 15, which means we're at the first view, which
77  //means we aren't able to go back any more views
78  back_push_button_->setEnabled(next_start_column);
79  forward_push_button_->setEnabled(true);//we can always move forward from the
80  //previous page, because in order for the back button to even be active,
81  //a viewable next page must be availible
82  current_start_column_ = next_start_column;
83  uint next_end_column = current_start_column_ - 15;
84  table_model_ = new VespucciTableModel(table_view_, table_model_->GetMatrix(), next_start_column-1);
